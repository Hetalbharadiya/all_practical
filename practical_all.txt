ConstructorInjectionExample.java

import java.util.*;

class StudentList {
    private List<String> students;

    public StudentList(List<String> students) {
        this.students = students;
    }

    public void showStudents() {
        System.out.println("Students (Constructor Injection):");
        for(String s : students) {
            System.out.println(s);
        }
    }
}

public class ConstructorInjectionExample {
    public static void main(String[] args) {
        List<String> list = new ArrayList<>();
        list.add("Hetal");
        list.add("Arin");
        list.add("John");

        StudentList obj = new StudentList(list);
        obj.showStudents();
    }
}


SetterInjectionExample.java

import java.util.*;

class StudentList {
    private List<String> students;

    public void setStudents(List<String> students) {
        this.students = students;
    }

    public void showStudents() {
        System.out.println("Students (Setter Injection):");
        for(String s : students) {
            System.out.println(s);
        }
    }
}

public class SetterInjectionExample {
    public static void main(String[] args) {
        List<String> list = new ArrayList<>();
        list.add("Hetal");
        list.add("Arin");
        list.add("John");

        StudentList obj = new StudentList();
        obj.setStudents(list);
        obj.showStudents();
    }
}
